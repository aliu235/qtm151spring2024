  • If "Overall > 90", classification = "Top Performer"

  • Extract the column "Overall" in the dataset
    (be carefeful to write the "O" capitalized)
  • Create a new variable called "player_classification"
    to assign each player to the correct classification

# Write your own code

def perform_status (Overall):
    if(Overall > 90):
        classification = "Top Performer"
    else: 
        classification = "Non Top-Performer"
    return classification

extracted_column = fifa['Overall']
fifa['player_classification'] = extracted_column.apply(perform_status)

display(fifa['player_classification'])

(d) Regression function

model = smf.ols(formula = 'Potential ~  Age', data = fifa)
results = model.fit()

b_list = results.params
print(b_list)

fifa["p_estimated"] = b_list[0] + b_list[1]  * fifa["Age"]

plt.scatter(x=fifa['Age'], y = fifa['Potential'])
plt.plot(fifa['Age'], fifa['p_estimated'], color = 'green')
